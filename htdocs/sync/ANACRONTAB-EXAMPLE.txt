# Exemple de configuration Anacron pour synchronisation Spider-VO
# 
# Copiez cette ligne dans votre fichier anacrontab sur Gandi
# Fichier : /srv/data/etc/cron/anacrontab (via Console d'urgence)
#          ou /lamp0/etc/cron/anacrontab (via SFTP)
#
# Synchronisation quotidienne Ã  3h du matin (tous les jours)
@daily 0 spider_vo_sync php -f /srv/data/web/vhosts/www.jdcauto.fr/htdocs/sync/spider_vo_sync.php

# Alternative : Toutes les 6 heures
# 6@hourly 0 spider_vo_sync php -f /srv/data/web/vhosts/www.jdcauto.fr/htdocs/sync/spider_vo_sync.php

# Alternative : Toutes les 12 heures
# 12@hourly 0 spider_vo_sync php -f /srv/data/web/vhosts/www.jdcauto.fr/htdocs/sync/spider_vo_sync.php
